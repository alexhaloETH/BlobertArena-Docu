119 LocalDataModel.ChoosenBlobert = PlayerDataModel.selectedPlayableCharacters[0].GetTokenId().Hex();
120 }
121
122 yield break; // Exit coroutine once successful
123 }
124
125 // Wait for 1 second before the next attempt
