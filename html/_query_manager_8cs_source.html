64 {
65 // Base template for the query with placeholders for entity name, arguments, and dynamic sections
66 string queryTemplate = @"
67 query {
68 {ENTITY_NAME} arguments {
69 edges {
70 node {
71 entity {
72 models {
73 __typename
74 ...DYNAMIC_SECTIONS...
75 }
76 }
77 }
78 }
79 totalCount
80 }
81 }
82 ";
83
84 // Combine all the section pieces into a single string
85 string combinedSections = string.Join(Environment.NewLine, sectionPieces);
86
87 // Replace placeholders in the query template
