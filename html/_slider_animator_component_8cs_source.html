281 private IEnumerator AnimateSlider(float targetValue, float duration, bool changeBackgroundColour = false, string format = "{0}/{1}")
282 {
283 float startValue = slider.value;
284 float elapsedTime = 0f;
285 isAnimating = true;
286
287 while (elapsedTime < duration)
288 {
289 elapsedTime += Time.deltaTime;
290 float currentValue = Mathf.Lerp(startValue, targetValue, elapsedTime / duration);
291 slider.value = currentValue;
292
293 if (changeBackgroundColour)
294 {
295 UpdateHealthBarColor();
296 }
297 // Update the TMP_Text component with the formatted string
298 if (textComponent != null)
299 {
300 textComponent.text = string.Format(format, Mathf.RoundToInt(currentValue), Mathf.RoundToInt(slider.maxValue));
301 }
302
303 yield return null;
304 }
305
306 isAnimating = false;
307 slider.value = targetValue; // Ensure it ends on the exact value
308
309 // Ensure the TMP_Text ends on the exact target value
310 if (textComponent != null)
311 {
312 textComponent.text = string.Format(format, Mathf.RoundToInt(targetValue), Mathf.RoundToInt(slider.maxValue));
