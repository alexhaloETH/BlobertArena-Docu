242 private async Task<string> EndOfBattleOutcomeString()
243 {
244 var healthOfPlayer = await CombatantDataModel.ManualQueryCombatantState(m_BattlePageController.StatsComponents[0].combatantId);
245 var healthOfEnemy = await CombatantDataModel.ManualQueryCombatantState(m_BattlePageController.StatsComponents[1].combatantId);
246
248 if (healthOfEnemy.health > 0 && healthOfPlayer.health > 0)
249 {
250 if (m_Outcome == Outcomes.PlayerWon)
251 {
252 return "ENEMY FORFEIT";
253 }
254
255 return "PLAYER FORFEIT";
256 }
257
258 try
259 {
260 // Get the appropriate combatant ID based on the outcome
261 var combatantId = m_Outcome == Outcomes.PlayerLost
262 ? m_BattlePageController.StatsComponents[1].combatantId
263 : m_BattlePageController.StatsComponents[0].combatantId;
264
265 // Query the round number
266 var roundNumberAwait = await AttackResult.GetAttackAndDamageResults(combatantId, (int)m_BattlePageController.RoundNumber);
267
268 // Return the outcome string
269 StringBuilder resultString = new();
270 resultString.Append($"LAST ATTACK WAS {WorldDataModel.AllAttacksDict[new FieldElement(roundNumberAwait.ItemId).Hex()].dojoName}");
271 resultString.Append($" DEALING {roundNumberAwait.Damage} DAMAGE.");
272 resultString.Append(m_Outcome == Outcomes.PlayerWon ? " YOUR OPPONENT HAS BEEN DEFEATED!" : " YOU HAVE BEEN DEFEATED!");
273 return resultString.ToString().ToUpper();
274 }
275 catch (Exception ex)
276 {
277 // Log the error or handle it as necessary
278 Console.WriteLine($"Error while calculating battle outcome: {ex.Message}");
279 return "Error"; // Return "Error" in case of any exceptions
280 }

283 private IEnumerator OutcomeExplanation()
284 {
285 SliderAnimatorComponent slider = null;
286
287 var task = EndOfBattleOutcomeString();
288 while (!task.IsCompleted)
289 {
290 yield return null;
291 }
292
293 var resultEndOfBattle = task.Result; // Use the result as needed
294
295 // Set it to wait//get the name of the attack
296 m_BattlePageFullLengthComponent.GetTypewriter().SetTextToType(resultEndOfBattle,
297 () => m_BattlePageFullLengthComponent.GetTypewriter().StartWaitingTextAnimation());
298
299 yield return new WaitUntil(() => m_BattlePageFullLengthComponent.GetTypewriter().IsTyping() == false);
300 yield return new WaitForSeconds(4f);
301
302 if (m_Outcome == Outcomes.PlayerWon)
303 {
304 m_BattlePageController.StatsComponents[1].HealthSlider.SetValue(0, 1f); // Starting the animation
305 slider = m_BattlePageController.StatsComponents[1].HealthSlider;
306 }
307 else
308 {
309 m_BattlePageController.StatsComponents[0].HealthSlider.SetValue(0, 1f); // Starting the animation
310 slider = m_BattlePageController.StatsComponents[0].HealthSlider;
311 }
312
313 yield return new WaitUntil(() => slider.isAnimating == false); // Once it has finished animation, show the outcome
314
315 yield return new WaitForSeconds(0.5f);
316
317 m_AnimationState = AnimStates.OutcomeExplenation;
318 }
