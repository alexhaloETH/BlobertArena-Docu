288 private ModelInstance[] LoadModels()
289 {
290 List<ModelInstance> models = new();
291
292 Assembly[] assemblies = AppDomain.CurrentDomain.GetAssemblies();
293
294 foreach (Assembly assembly in assemblies)
295 {
296 var modelTypes = assembly.GetTypes()
297 .Where(t => typeof(ModelInstance).IsAssignableFrom(t) && !t.IsAbstract);
298
299 foreach (Type modelType in modelTypes)
300 {
301 models.Add((ModelInstance)gameObject.AddComponent(modelType));
302 }
303 }
304
305 return models.ToArray();
