113 //Fetch entities from the world untill none are left
114 public async Task<List<Entity>> FetchEntitiesByBatch(Query query)
115 {
116 List<Entity> allEntities = new List<Entity>();
117 uint batchSize = query.limit;
118 bool hasMoreEntities = true;
119
120 while (hasMoreEntities)
121 {
122 List<Entity> entities;
123#if UNITY_WEBGL && !UNITY_EDITOR
124 entities = await wasmClient.Entities(query);
125#else
126 entities = await Task.Run(() => toriiClient.Entities(query));
127#endif
128
129 allEntities.AddRange(entities);
130
131 if (entities.Count < batchSize)
132 {
133 hasMoreEntities = false;
134 }
135 else
136 {
137 query.offset += batchSize;
138 }
139 }
140
141 return allEntities;
142 }
