48 public TextureFunctionInput(Texture texture, string name, int startIndex, int length)
49 {
50 this.Texture = texture;
51 this.Name = name;
52 this.StartIndex = startIndex;
53 this.Length = length;
54
55 var allSprites = Resources.LoadAll<Sprite>(name).ToList();
56
57 if (startIndex >= 0 && startIndex + length <= allSprites.Count)
58 {
59 Sprites = allSprites.GetRange(startIndex, length);
60 }
61 else
62 {
63 Debug.LogWarning("Requested sprite range is out of bounds. Assigning an empty list.");
64 Sprites = new List<Sprite>(); // Assign an empty list if out of bounds
65 }
66 }

73 public TextureFunctionInput(List<Sprite> sprites, int length)
74 {
75 this.Texture = null;

78 this.Texture = sprites[0].texture;

83 }
84
85 this.Name = "";
86 this.StartIndex = 0;
87 this.Length = length;

359 private void Update()
360 {
361 if (_paused)
362 { return; }
363
364 // Check if there's an animation in the queue
365 if (QueueOfPlayOverAnims.Count > 0)
366 {
367 var (textureInputs, drawOver, callback) = QueueOfPlayOverAnims.Peek();
368
369 if (drawOver && textureInputs.Count == 1 && textureInputs[0].Sprites.Count > 0)
370 {
371 // Animate the overlay image with the sprites in textureInputs[0]
372 _timeSinceLastFrame += Time.deltaTime;
373
374 if (_timeSinceLastFrame >= Speed)
375 {
376 // Update the overlay image with the current frame
377 EffectOverlayImage.sprite = textureInputs[0].Sprites[currentFrameDrawover % textureInputs[0].Sprites.Count];
378 EffectOverlayImage.enabled = true;
379
380 // Move to the next frame
381 currentFrameDrawover++;
382 CheckForCallback(currentFrameDrawover, callback);
383 _timeSinceLastFrame = 0f;
384
385 // Check if we reached the end of the animation
386 if (currentFrameDrawover >= textureInputs[0].Sprites.Count)
387 {
388 // Reset overlay and dequeue animation after it finishes
389 QueueOfPlayOverAnims.Dequeue();
390 currentFrameDrawover = 0;
391 EffectOverlayImage.enabled = false; // Hide overlay after animation
392 }
393 }
394 return; // Skip idle animation while drawing overlay
395 }
396 else if (!drawOver)
397 {
398 // Stop idle animation and play queued animation on main images
399 _timeSinceLastFrame += Time.deltaTime;
400
401 if (_timeSinceLastFrame >= Speed)
402 {
403 for (int i = 0; i < textureInputs.Count; i++)
404 {
405 var imageComponent = ImageObjects[i].GetComponent<Image>();
406 imageComponent.sprite = textureInputs[i].Sprites[currentFrameDrawover % textureInputs[i].Sprites.Count];
407 imageComponent.SetAllDirty();
408 }
409
410 currentFrameDrawover++;
411 CheckForCallback(currentFrameDrawover, callback);
412 _timeSinceLastFrame = 0f;
413
414 // Check if the main animation is over
415 if (currentFrameDrawover >= textureInputs[0].Sprites.Count)
416 {
417 QueueOfPlayOverAnims.Dequeue(); // Remove current animation
418 currentFrameDrawover = 0; // Reset to start idle animation
419 EffectOverlayImage.enabled = false; // Hide overlay if it was active
420 }
421 }
422 return; // Skip idle animation while playing queue animation
423 }
424 }
425 else
426 {
427 EffectOverlayImage.enabled = false; // Ensure overlay is hidden if no queue item
428 }
429
430 // Regular idle animation if no overlay/queue item
431 if (IdleTextureFunctionInput.Count == 0)
432 return;
433
434 _timeSinceLastFrame += Time.deltaTime;
435 if (_timeSinceLastFrame >= Speed)
436 {
437 for (int i = 0; i < IdleTextureFunctionInput.Count; i++)
438 {
439 var textureInput = IdleTextureFunctionInput[i];
440 if (textureInput.Sprites.Count == 0)
441 continue;
442
443 var imageComponent = ImageObjects[i].GetComponent<Image>();
444 imageComponent.sprite = textureInput.Sprites[_currentFrame % textureInput.Sprites.Count];
445 imageComponent.SetAllDirty();
446 }
447 _currentFrame++;
448 _timeSinceLastFrame = 0f;
449 }

451 void CheckForCallback(int frameNum, List<TweenCallbackSystem> callbackSystem)
452 {
453 if (callbackSystem == null)
454 return;
455
456 if (callbackSystem.Count == 0)
457 return;
458
459 if (frameNum != callbackSystem[0].frameNum)
460 return;
461
462 callbackSystem[0].functionCallback();
463 callbackSystem.RemoveAt(0);
464 }
465 }
