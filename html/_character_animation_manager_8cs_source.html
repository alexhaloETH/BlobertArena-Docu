19 public TextureFunctionInput(Texture texture, string name, int startIndex, int length)
20 {
21 this.Texture = texture;
22 this.Name = name;
23 this.StartIndex = startIndex;
24 this.Length = length;
25
26 var allSprites = Resources.LoadAll<Sprite>(name).ToList();
27
28 if (startIndex >= 0 && startIndex + length <= allSprites.Count)
29 {
30 Sprites = allSprites.GetRange(startIndex, length);
31 }
32 else
33 {
34 Debug.LogWarning("Requested sprite range is out of bounds. Assigning an empty list.");
35 Sprites = new List<Sprite>(); // Assign an empty list if out of bounds
36 }
37 }

40 {
41 this.Texture = null;
42 try

44 this.Texture = sprites[0].texture;

51 this.Name = "";

53 this.Length = length;
54

262 private void Update()
263 {
264 if (_paused)
265 { return; }
266
267 // Check if there's an animation in the queue
268 if (QueueOfPlayOverAnims.Count > 0)
269 {
270 var (textureInputs, drawOver, callback) = QueueOfPlayOverAnims.Peek();
271
272 //if (textureInputs == null)
273 // Debug.Log("texture null");
274
275 //if (drawOver == null)
276 // Debug.Log("draw null");
277
278 //if (callback == null)
279 // Debug.Log("callback null");
280
281 if (drawOver && textureInputs.Count == 1 && textureInputs[0].Sprites.Count > 0)
282 {
283 // Animate the overlay image with the sprites in textureInputs[0]
284 _timeSinceLastFrame += Time.deltaTime;
285
286 if (_timeSinceLastFrame >= Speed)
287 {
288 // Update the overlay image with the current frame
289 EffectOverlayImage.sprite = textureInputs[0].Sprites[currentFrameDrawover % textureInputs[0].Sprites.Count];
290 EffectOverlayImage.enabled = true;
291
292 // Move to the next frame
293 currentFrameDrawover++;
294 CheckForCallback(currentFrameDrawover, callback);
295 _timeSinceLastFrame = 0f;
296
297 // Check if we reached the end of the animation
298 if (currentFrameDrawover >= textureInputs[0].Sprites.Count)
299 {
300 // Reset overlay and dequeue animation after it finishes
301 QueueOfPlayOverAnims.Dequeue();
302 currentFrameDrawover = 0;
303 EffectOverlayImage.enabled = false; // Hide overlay after animation
304 }
305 }
306 return; // Skip idle animation while drawing overlay
307 }
308 else if (!drawOver)
309 {
310 // Stop idle animation and play queued animation on main images
311 _timeSinceLastFrame += Time.deltaTime;
312
313 if (_timeSinceLastFrame >= Speed)
314 {
315 for (int i = 0; i < textureInputs.Count; i++)
316 {
317 var imageComponent = ImageObjects[i].GetComponent<Image>();
318 imageComponent.sprite = textureInputs[i].Sprites[currentFrameDrawover % textureInputs[i].Sprites.Count];
319 imageComponent.SetAllDirty();
320 }
321
322 currentFrameDrawover++;
323 CheckForCallback(currentFrameDrawover, callback);
324 _timeSinceLastFrame = 0f;
325
326 // Check if the main animation is over
327 if (currentFrameDrawover >= textureInputs[0].Sprites.Count)
328 {
329 QueueOfPlayOverAnims.Dequeue(); // Remove current animation
330 currentFrameDrawover = 0; // Reset to start idle animation
331 EffectOverlayImage.enabled = false; // Hide overlay if it was active
332 }
333 }
334 return; // Skip idle animation while playing queue animation
335 }
336 }
337 else
338 {
339 EffectOverlayImage.enabled = false; // Ensure overlay is hidden if no queue item
340 }
341
342 // Regular idle animation if no overlay/queue item
343 if (IdleTextureFunctionInput.Count == 0)
344 return;
345
346 _timeSinceLastFrame += Time.deltaTime;
347 if (_timeSinceLastFrame >= Speed)
348 {
349 for (int i = 0; i < IdleTextureFunctionInput.Count; i++)
350 {
351 var textureInput = IdleTextureFunctionInput[i];
352 if (textureInput.Sprites.Count == 0)
353 continue;
354
355 var imageComponent = ImageObjects[i].GetComponent<Image>();
356 imageComponent.sprite = textureInput.Sprites[_currentFrame % textureInput.Sprites.Count];
357 imageComponent.SetAllDirty();
358 }
359 _currentFrame++;
360 _timeSinceLastFrame = 0f;
361 }
