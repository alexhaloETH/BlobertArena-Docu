185 private IEnumerator PingMatchMakingServerCoroutine()
186 {
187 while (true)
188 {
189 //this here needs a listener
190 if (!m_OpenToInvite.IsOpenToInvite())
191 {
192 Debug.Log("Open to invite is false");
193 yield return new WaitForSeconds(180f); // Avoid busy-waiting; check again after 1 second.
194 continue;
195 }
196
197 if (LocalDataModel.CurrentAccount == null)
198 {
199 yield return new WaitForSeconds(180f); // Avoid busy-waiting; check again after 1 second.
200 continue;
201 }
202
203 Debug.Log("i am requstiiong to have an invite");
204 var pingTask = LogicLayer.BackEndServer.QuickFightModel.PingQuickMatchEndPointAsync(
205 LocalDataModel.CurrentAccount.Address.Hex(), LocalDataModel.Username);
206
207 while (!pingTask.IsCompleted)
208 {
209 yield return null; // Wait for the async task to complete.
210 }
211
212 if (pingTask.Exception != null)
213 {
214 Debug.LogError($"Error during PingMatchMakingServerCoroutine: {pingTask.Exception.Message}");
215 yield return new WaitForSeconds(60f); // Wait before retrying in case of an error.
216 }
217 else
218 {
219 yield return new WaitForSeconds(60f); // Wait for 60 seconds before the next ping.
220 }
221 }
