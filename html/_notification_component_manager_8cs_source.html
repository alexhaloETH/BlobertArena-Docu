280 // Utility function to show the next notification
281 private void ShowNextNotification()
282 {
283 var notificationData = _notificationQueue.First?.Value;
284
285 if (notificationData == null)
286 {
287 return;
288 }
289
290 _notificationObject.SetActive(true);
291
292 foreach (var button in notificationData.Value.ButtonTypes)
293 {
294 var buttonPrefab = Instantiate(ButtonPrefabs[(int)button.ButtonType], _buttonParent.transform);
295
296 var baseButton = buttonPrefab.GetComponent<BaseButton>();
297 baseButton.GetText().text = button.Message;
298 baseButton.onClickCalls = button.Callbacks;
299 }
300
301 _notiText.text = notificationData.Value.Message;
302
303 _notibackground.sprite = BackgroundColours[(int)notificationData.Value.ColourOfBackground];
304
305 if (notificationData.Value.BlobId != null)
306 {
307 _notiCharAnimator.gameObject.SetActive(true);
308
309 var blobData = WorldDataModel.AllPlayableCharactersInTheGame[notificationData.Value.BlobId.Hex()];
310 if (blobData.GetNftType() == BlobertTypes.NFTTypes.AMMABlobs)
311 {
312 _notiCharAnimator.InitilizeComponentQuick(WorldDataModel.AllPlayableCharactersInTheGame[notificationData.Value.BlobId.Hex()], AmmaAnimationController.QuickIdleNoMaskR);
313 }
314 else
315 {
316 _notiCharAnimator.InitilizeComponentQuick(WorldDataModel.AllPlayableCharactersInTheGame[notificationData.Value.BlobId.Hex()], BlobertAnimationController.IdleDefaultFrontR);
317 }
318 }
319 else
320 {
321 _notiCharAnimator.gameObject.SetActive(false);
322 }
323
324 _notificationTimer = _timerPerNotification;
