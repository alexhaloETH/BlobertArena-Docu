222 public static async Task<FieldElement> InvokeViaLocalAccount(Account account)
223 {
224 var calls = new List<dojo.Call>();
225
226 foreach (var structInstance in callsList)
227 {
228 var calldata = new List<dojo.FieldElement>();
229
230 var endpointData = GetFieldsFromStruct(structInstance, out var fields);
231
232 foreach (var field in fields)
233 {
234 var fieldValue = field.GetValue(structInstance);
235 ProcessStructValues(fieldValue, calldata);
236 }
237
238 calls.Add(new dojo.Call
239 {
240 calldata = calldata.ToArray(),
241 selector = endpointData.functionName,
242 to = endpointData.addressOfSystem.Inner
243 });
244 }
245
246 return await account.ExecuteRaw(calls.ToArray());
247 }

327 public static void ProcessStructValues(object fieldValue, List<dojo.FieldElement> callData)
328 {
329 switch (fieldValue)
330 {
331 case BigInteger bi:
332 callData.Add(new FieldElement(bi).Inner);
333 break;
334
335 case byte by:
336 callData.Add(new FieldElement(by.ToString("X")).Inner);
337 break;
338
339 case UInt16 u16:
340 callData.Add(new FieldElement(u16.ToString("X")).Inner);
341 break;
342
343 case UInt32 u32:
344 callData.Add(new FieldElement(u32.ToString("X")).Inner);
345 break;
346
347 case UInt64 u64:
348 callData.Add(new FieldElement(u64.ToString("X")).Inner);
349 break;
350
351 case FieldElement fe:
352 callData.Add(fe.Inner);
353 break;
354
355 case String str:
356 callData.Add(new FieldElement(str).Inner);
357 break;
358
359 case bool bo:
360 Debug.LogError("bool has yet to be implemented, message alex");
361 break;
362
363 default:
364 if (fieldValue != null && fieldValue.GetType().IsEnum)
365 {
366 // Convert enum to its underlying integer value
367 var enumValue = Convert.ToInt32(fieldValue);
368 callData.Add(new FieldElement(enumValue.ToString("X")).Inner);
369 }
370 else if (fieldValue != null && fieldValue.GetType().IsValueType)
371 {
372 // Handle custom struct by recursively processing its fields
373 var nestedFields = fieldValue.GetType().GetFields();
374 foreach (var nestedField in nestedFields)
375 {
376 var nestedFieldValue = nestedField.GetValue(fieldValue);
377 ProcessStructValues(nestedFieldValue, callData);
378 }
379 }
380 else
381 {
382 try
383 {
384 callData.Add(new FieldElement(((int)fieldValue).ToString("X")).Inner);
385 }
386 catch
387 {
388 Debug.LogError($"Unsupported field type: {fieldValue?.GetType().Name}");
389 }
390 }
391
392 break;
393 }
