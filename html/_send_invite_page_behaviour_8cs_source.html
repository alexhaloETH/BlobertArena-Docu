93
94 try
95 {
96 _inviteButton.LockButton();

98 string resolvedAddress;

102 if (inputText.Length == 66 && inputText.StartsWith("0x"))
103 {
104 resolvedAddress = inputText;
105 isSuccess = true;
106 }
107 else
108 {
109 (resolvedAddress, isSuccess) = await BlobertUtils.QuerySNSIds(inputText, (int)_walletSelector.GetCurrentWallet());
110 Debug.Log($"Resolved address: {resolvedAddress}, isSuccess: {isSuccess}");
111 }
112
113 if (!isSuccess || resolvedAddress == "0x0")
114 {
115 UiReferenceManager.instance.notificationComponent.CreateErrorNotification("THIS PLAYER DOES NOT EXIST", true);
116 throw new Exception("Invalid address or username resolution failed");
117 }
118
119 _otherPlayerAddressInputfield.text = resolvedAddress;
120
121 if (resolvedAddress == LocalDataModel.CurrentAccount.Address.Hex())
122 {
123 UiReferenceManager.instance.notificationComponent.CreateErrorNotification("THIS PLAYER DOES NOT EXIST", true);
124 throw new Exception("You can't invite yourself");
125 }
126
127 var attackStruct =
128 WorldDataModel.GenerateAttackStructFromCharacter(new FieldElement(LocalDataModel.ChoosenBlobert));
129
130 // Construct and send invite
131 var inviteStruct = PvpActionsContract.SendInvite(
132 receiver: new FieldElement(resolvedAddress),
133 collectionAddress: DojoEntitiesStorage.worldManagerData.arcadeBlobertActions,
134 tokenId: new FieldElement(LocalDataModel.ChoosenBlobert),
135 attacks: attackStruct,
136 phaseTime: 60,
137 address: DojoEntitiesStorage.worldManagerData.pvpActions
138 );
139
140 DojoContractCommunication.AddToCallsList(inviteStruct, true);
141 await DojoContractCommunication.InvokeContract(gameObject.name, "OnChainCallbackSendInvite", account: LocalDataModel.CurrentAccount);
142 }
143 catch (Exception ex)
144 {
145 UiReferenceManager.instance.notificationComponent.CreateErrorNotification("ERROR OCCURED INVITING PLAYER", true);
146 _otherPlayerAddressInputfield.text = "";
