106 (resolvedAddress, isSuccess) = await StarknetUtils.SnsUtils.QuerySNSIds(inputText, (int)_walletSelector.GetCurrentWallet());
107 Debug.Log($"Resolved address: {resolvedAddress}, isSuccess: {isSuccess}");
108 }
109
110 if (!isSuccess || resolvedAddress == "0x0")
111 {
112 UiReferenceManager.instance.notificationComponent.CreateErrorNotification("THIS PLAYER DOES NOT EXIST", true);
113 }
114
115 _otherPlayerAddressInputfield.text = resolvedAddress;
116
117 if (resolvedAddress == LocalDataModel.CurrentAccount.Address.Hex())
118 {
119 UiReferenceManager.instance.notificationComponent.CreateErrorNotification("THIS PLAYER DOES NOT EXIST", true);

121
122 var attackStruct =
123 WorldDataModel.GenerateAttackStructFromCharacter(new FieldElement(LocalDataModel.ChoosenBlobert));
124
125 // Construct and send invite
126 var inviteStruct = PvpActionsContract.SendInvite(
127 receiver: new FieldElement(resolvedAddress),
128 collectionAddress: DojoEntitiesStorage.worldManagerData.arcadeBlobertActions,
129 tokenId: new FieldElement(LocalDataModel.ChoosenBlobert),
130 attacks: attackStruct,
131 phaseTime: 60,
132 address: DojoEntitiesStorage.worldManagerData.pvpActions
133 );
134
135 DojoContractCommunication.AddToCallsList(inviteStruct, true);
136 await DojoContractCommunication.InvokeContract(gameObject.name, "OnChainCallbackSendInvite", account: LocalDataModel.CurrentAccount);
137 }
138 catch (Exception ex)
139 {
140 UiReferenceManager.instance.notificationComponent.CreateErrorNotification("ERROR OCCURED INVITING PLAYER", true);
141 _otherPlayerAddressInputfield.text = "";
