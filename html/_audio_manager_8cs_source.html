275 public void PlayMusic(AudioEffects.Music clip, AudioEffects.Music loopAudio = AudioEffects.Music.NONE)
276 {
277 StopMusic();
278 m_CurrentClip = clip;
279 if (clip == AudioEffects.Music.NONE) { return; }
280 m_MusicClips[(int)clip].Source.Play();
281 m_WaiterCoroutine = StartCoroutine(WaitAndPlayAudio(loopAudio));
282 }

338 private IEnumerator WaitAndPlayAudio(AudioEffects.Music nextClip)
339 {
340 yield return new WaitUntil(() => !m_MusicClips[(int)m_CurrentClip].Source.isPlaying);
341 m_CurrentClip = nextClip;
342
343 if (nextClip == AudioEffects.Music.NONE)
344 {
345 yield break;
346 }
347
348 m_MusicClips[(int)nextClip].Source.Play();
349 yield return null;
350 }
