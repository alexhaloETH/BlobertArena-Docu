82 public static AnimationDataTraitBlobert DecodeAnimationData(int encoded)
83 {
84 // Extract the data using bit-masking and shifting
85 AnimationDirection direction = (AnimationDirection)(encoded & 0b1);
86 AnimationFace side = (AnimationFace)((encoded >> 1) & 0b1);
87 AnimationState state = (AnimationState)((encoded >> 2) & 0b11);
88 BehaviourState behaviour = (BehaviourState)((encoded >> 4) & 0b111);
89
90 return new AnimationDataTraitBlobert(direction, side, state, behaviour);

372 public static T[] GetSubArray<T>(T[] array, int startIndex, int length)
373 {
374 if (array == null)
375 throw new ArgumentNullException(nameof(array));
376
377 if (startIndex < 0 || startIndex > array.Length)
378 throw new ArgumentOutOfRangeException(nameof(startIndex), "Start index must be within the bounds of the array.");
379
380 if (length < 0 || (startIndex + length) > array.Length)
381 throw new ArgumentOutOfRangeException(nameof(length), "Length must be non-negative and the range must be within the bounds of the array.");
382
383 T[] result = new T[length];
384 Array.Copy(array, startIndex, result, 0, length);
385 return result;
386 }
