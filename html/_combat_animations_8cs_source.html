92 public bool PlayAnimation(AnimationTypes animationType, bool forcePlay = false)
93 {
94 if (forcePlay) { StopAnimation(); } // Forcing to stop the animation that is playing currently
95 else if (LeanTween.isTweening(gameObject)) { return false; } // If an animations was playing, return false
96
97 m_CurrentAnimation = animationType;
98
99 switch (animationType)
100 {
101 case AnimationTypes.Idle:
102 // Do nothing, this is the standard state
103 break;
104
105 case AnimationTypes.Stun:
106 {
107 gameObject.transform.localPosition -= gameObject.transform.right * (m_StunIntensity * 0.5f);
108 m_StunnedAnim = LeanTween.moveLocalX(gameObject, m_StunIntensity, 1.25f)
109 .setEaseInOutBack()
110 .setLoopPingPong();
111 }
112 break;
113
114 case AnimationTypes.Shaken:
115 {
116 gameObject.transform.localPosition -= gameObject.transform.right * (m_ShakeIntensity * 0.5f);
117 m_ShakeAnim = LeanTween.moveLocalX(gameObject, m_ShakeIntensity, m_ShakeSpeed)
118 .setLoopPingPong()
119 .setRepeat(SHAKE_COUNT * 2)
120 .setOnComplete(() => ResetPosition());
121
122 PlayExplosionEffect();
123 }
124 break;
125
126 case AnimationTypes.Angry:
127 {
128 m_AngryAnim = LeanTween.moveLocalY(gameObject, m_AngryIntensity, m_TantrumSpeed)
129 .setEaseInBack()
130 .setEaseOutQuad()
131 .setLoopPingPong()
132 .setRepeat(JUMP_COUNT * 2)
133 .setOnComplete(() => ResetPosition());
134 }
135 break;
136 }
137 return true;
138 }
