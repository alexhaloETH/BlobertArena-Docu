472 public static async Task<List<BattlePageController.AttackAvailability>> LoadStateOfAttacksForCombatant(FieldElement combatant, uint currentRound = 1)
473 {
474 var returnData = new List<BattlePageController.AttackAvailability>();
475 var attacks = await CombatantDataModel.GetAvailableAttacksOfTheCombatant(combatant);
476
477 foreach (var (attackId, lastRoundUsed) in attacks)
478 {
479 var attackData = WorldDataModel.AllAttacksDict[attackId.Hex()];
480 var attackCooldown = attackData.cooldown;
481 var availability = false;
482 var roundUntil = 0;
483
484 // If cooldown is 0, attack is always available
485 if (attackCooldown == 0)
486 {
487 availability = true;
488 }
489 else if (lastRoundUsed == 0 || (lastRoundUsed + attackCooldown <= currentRound))
490 {
491 availability = true;
492 }
493 else
494 {
495 roundUntil = (int)(lastRoundUsed + attackCooldown - currentRound);
496 }
497
498 returnData.Add(new BattlePageController.AttackAvailability
499 {
500 Attack = attackData,
501 Availability = availability,
502 RoundUntil = roundUntil
503 });
504 }
505
506 // Sort attacks using the WorldDataModel's sorting method
507 var sortedAttacks = WorldDataModel.SortAttacks(returnData.Select(tuple => tuple.Attack).ToList());
508
509 var sortedReturnData = sortedAttacks.Select(sortedAttack =>
510 returnData.First(tuple => tuple.Attack == sortedAttack)).ToList();
511
512 return sortedReturnData;
513 }
