154 public void SetBattleState(BattleState newState)
155 {
156 if (gameHasEnded)
157 {
158 var state = EvaluteOnChainState();
159
160 gameHasEnded = false;
161
162 if (state == BattleState.EndOfBattle)
163 {
164 HardSetBattleState(BattleState.EndOfBattle);
165 return;
166 }
167 }
168
169 Debug.Log($"the new state is {newState} and the current one is {CurrentState}");
170 if (CurrentState == newState)
171 return;
172
173 if (newState == BattleState.EndOfBattle)
174 {
175 //ToggleBattleUi(false);
176 AcceptedStateChange(newState);
177 return;
178 }
179
180 if ((CurrentState is BattleState.RevealAttack or BattleState.WaitingForEnemyToReveal or BattleState.RunRoundCallWait)
181 && newState == BattleState.CommitAttack)
182 {
183 AcceptedStateChange(BattleState.RoundOutcomeFirst);
184 _dojoData = null;
185 Debug.Log($"Transitioned to RoundOutcomeSelf as chain reset was detected. Current State: {CurrentState}");
186 return;
187 }
188
189 if (CurrentState == BattleState.WaitingForEnemyToReveal && newState == BattleState.RevealAttack)
190 {
191 AcceptedStateChange(BattleState.RevealAttack);
192 return;
193 }
194
195 if (newState == BattleState.CommitAttack)
196 {
197 for (int i = 0; i < _salts.Length; i++)
198 {
199 _salts[i] = new FieldElement(StarknetUtils.StarknetHashOperationsUtils.GetRandomFeltInHex());
200 }
201 }
202
203 if (newState <= CurrentState) return;
204
205 AcceptedStateChange(newState);
206 }

209 {
210 // Check if the new state is the same as the current state
211 if (CurrentState == newState)
212 return;
213
214 if (newState == BattleState.EndOfBattle)
215 {
216 //if the current state is the outcome states we are not allowed to go into the new state
217 if (CurrentState is BattleState.RoundOutcomeFirst or BattleState.RoundOutcomeSecond)
218 {
219 Debug.Log("Cannot transition to EndOfBattle from RoundOutcome states");
220 gameHasEnded = true;
221 return;
222 }
223
224 //ToggleBattleUi(false);

227 }
228

230 {
231 for (int i = 0; i < _salts.Length; i++)
232 {
233 _salts[i] = new FieldElement(StarknetUtils.StarknetHashOperationsUtils.GetRandomFeltInHex());
234 }
235 }
236
237 // Set the new state and invoke the event
238 AcceptedStateChange(newState);
239 Debug.Log("this is a call from the HardSetBattleState function and the state right now is " + CurrentState);

596 public BattleState EvaluteOnChainState()
597 {
598 var currentStatus = BattleState.CommitAttack;
599
600 if (CombatState.phase is Phase.Ended)
601 {
602 return BattleState.EndOfBattle;
603 }
604
605 var commitmentState = CombatantDataModel.BitsToWaitState(CombatantDataModel.GetCommitmentState(Commitments[0], Commitments[1]));
606 var plannedAttacksState = CombatantDataModel.BitsToWaitState(CombatantDataModel.GetPlannedAttackState(PlannedAttacks[0], PlannedAttacks[1]));
607
608 if (commitmentState == CombatantDataModel.WaitState.BothDone)
609 {
610 if (plannedAttacksState == CombatantDataModel.WaitState.BothDone)
611 {
612 return BattleState.RunRoundCallWait;
613 }
614 if (isOwner) //if i am the owner i want to wait until the other guy as done something
615 {
616 if (plannedAttacksState is CombatantDataModel.WaitState.EnemyWaiting)
617 {
618 return BattleState.RevealAttack;
619 }
620
621 if (plannedAttacksState is CombatantDataModel.WaitState.BothNothing or CombatantDataModel.WaitState.SelfWaiting)
622 {
623 return BattleState.WaitingForEnemyToReveal;
624 }
625 }
626 else
627 {
628 if (plannedAttacksState is CombatantDataModel.WaitState.EnemyWaiting or CombatantDataModel.WaitState.BothNothing)
629 {
630 return BattleState.RevealAttack;
631 }
632
633 if (plannedAttacksState is CombatantDataModel.WaitState.SelfWaiting)
634 {
635 return BattleState.WaitingForEnemyToReveal;
636 }
637 }
638 }
639
640 if (commitmentState == CombatantDataModel.WaitState.SelfWaiting)
641 {
642 return BattleState.WaitingForEnemyToCommit;
643 }
644
645 if (commitmentState is CombatantDataModel.WaitState.BothNothing or CombatantDataModel.WaitState.EnemyWaiting)
646 {
647 return BattleState.CommitAttack;
648 }
649
650 Debug.Log("this is at the end of the evaluating chain " + currentStatus);
651 return currentStatus;
652 }
